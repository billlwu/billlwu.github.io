def get_vol(s, logr):
    var = 0.0
    lam = 0.0
    var_list = []
    for r in logr:
        lam = lam*s + 1
        var = (1-1.0/lam)*var + (1.0/lam)*r**2
        var_list.append(var)
    return [np.sqrt(x) for x in var_list]
